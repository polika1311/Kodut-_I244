# tabeli loomine
 
CREATE TABLE ppopova_loomaaed (
        id integer PRIMARY KEY auto_increment,
        nimi varchar(100) NOT NULL,
        vanus integer DEFAULT 1,
        liik varchar(250) NOT NULL,
        puuri_nr  integer DEFAULT 1
);
